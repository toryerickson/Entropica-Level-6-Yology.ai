\subsection{The Two Primary Health Signals}

Operators monitor two key metrics:

\begin{enumerate}
    \item \textbf{$\Delta S$ (Behavioral Entropy)} --- Per-capsule (Vol.~I \S3). Measures how ``strange'' an individual capsule is behaving relative to its baseline. Threshold: $\tau$ (role-dependent, typically 0.3--0.5).
    
    \item \textbf{SCI (Swarm Coherence Index)} --- Per-trunk/branch (Vol.~II \S3.2). Measures how ``aligned'' a population is behaving collectively. Range: 0--1. Threshold: $\theta_{SCI}$ (deployment-dependent, typically 0.65--0.85).
\end{enumerate}

\begin{notebox}
\textbf{Key Insight:} A swarm can have all capsules with $\Delta S < \tau$ (all individually ``safe'') while $SCI < \theta_{SCI}$ (collectively drifting). Both signals are necessary.
\end{notebox}

\section{Standing Up an EFM Forest}

\subsection{Pre-Deployment Checklist}

\begin{enumerate}
    \item \textbf{Define Vault Commandments} --- Immutable safety rules. Once deployed, cannot be changed without full system rebuild.
    
    \item \textbf{Configure Genesis Parameters} --- Role archetypes, $\tau$/$\lambda$ defaults, permission matrices.
    
    \item \textbf{Provision Capsules} --- Deploy initial capsule fleet with Genesis Protocol. Verify each capsule's Vault Hash Chain links to system root.
    
    \item \textbf{Deploy Arbiter Set} --- Minimum 5 Judicial Capsules for $f=2$ Byzantine tolerance. Configure d-CAM quorum parameters.
    
    \item \textbf{Initialize d-CTM} --- Distributed ledger storage. Verify append-only property and ZK-SP integration.
    
    \item \textbf{Configure Gardener Keys} --- Generate and securely store Gardener override keys. Test override on non-critical verdict.
    
    \item \textbf{Enable SCI Computation} --- Configure $d_{behav}$ components, sampling strategy (if $>10^4$ capsules), and alert thresholds.
\end{enumerate}

% ============ SECTION 4 ============
\section{Response Playbooks}

\subsection{Entropy Spike (S2--S3 Severity)}

\begin{playbookbox}[Playbook: High $\Delta S$ on Individual Capsule]
\textbf{Trigger:} $\Delta S \geq \tau$ for capsule $C$, Reflex halt issued.

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Inspect History:} Query d-CTM for $C$'s recent action log. Identify which $\Delta S$ component spiked (action, resource, goal).
    
    \item \textbf{Classify:}
    \begin{itemize}
        \item \textit{Expected workload change} $\rightarrow$ Adjust $\tau$ via Threshold Governance (Vol.~II \S2.6)
        \item \textit{Legitimate anomaly} $\rightarrow$ Leave halt in place, monitor
        \item \textit{Misconfiguration} $\rightarrow$ Fix configuration, restart capsule
        \item \textit{Potential compromise} $\rightarrow$ Escalate to Arbiter for QUARANTINE
    \end{itemize}
    
    \item \textbf{Document:} Log decision rationale in d-CTM via Gardener annotation.
\end{enumerate}

\textbf{Escalation:} If pattern repeats across multiple capsules, investigate swarm-level cause (see SCI Dip playbook).
\end{playbookbox}

\subsection{SCI Below Threshold}

\begin{playbookbox}[Playbook: SCI Dip Below $\theta_{fork}$]
\textbf{Trigger:} $SCI < \theta_{fork}$ for $> T_{debounce}$ ticks.

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Identify Divergence:} Use DDI breakdown to find which capsule subset is drifting. Check behavioral distance matrix.
    
    \item \textbf{Review Arbiter Verdicts:} Query recent FORK proposals. If Arbiter already deliberating, wait for verdict.
    
    \item \textbf{Validate Fork Decision:}
    \begin{itemize}
        \item If FORK approved: Verify branch created correctly, SCI improved per-group
        \item If FORK rejected: Review DCG rationale; consider manual Gardener review
    \end{itemize}
    
    \item \textbf{Post-Fork Monitoring:} Track branch stability. Prepare for eventual Merge if branch proves valuable.
\end{enumerate}

\textbf{Escalation:} If SCI continues dropping post-fork, investigate deeper (possible adversarial drift or systemic issue).
\end{playbookbox}

\subsection{Orphan Detection}

\begin{playbookbox}[Playbook: Orphan Capsules Detected]
\textbf{Trigger:} \texttt{orphan\_count} $> 0$.

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Verify Protocol Active:} Confirm Orphan Protocol is running (capsules should be in QUARANTINE).
    
    \item \textbf{Check Recovery Attempts:} Query d-CTM for re-sync attempts. Are compatible trunks being tried?
    
    \item \textbf{Monitor Timeout:} If approaching $N_{orphan\_timeout}$, decide:
    \begin{itemize}
        \item \textit{Recoverable} $\rightarrow$ Manually assign to trunk via Gardener action
        \item \textit{Unrecoverable} $\rightarrow$ Allow PURGE, review forensics
    \end{itemize}
    
    \item \textbf{Post-Purge Review:} Examine d-CTM entry for PURGE. Verify compliance with E2 Purge Justification constraint (Vol.~II \S3.9.5).
\end{enumerate}

\textbf{Escalation:} Multiple orphans simultaneously may indicate trunk corruption or network partition.
\end{playbookbox}

\subsection{Catastrophic Failure: Arbiter Unavailability}

\begin{playbookbox}[Runbook 1: ALL Arbiters Unavailable]
\textbf{Trigger:} Arbiter availability $< 20\%$ or all Arbiters unreachable.

\textbf{Severity:} S4 (CRITICAL)

\textbf{Immediate Actions:}
\begin{enumerate}
    \item \textbf{Verify Scope:} Confirm this is not a monitoring failure. Check network connectivity to Arbiter nodes.
    
    \item \textbf{Emergency Spawn:} If truly unavailable, spawn emergency Arbiters from standby pool:
    \begin{verbatim}
    gardener spawn-arbiters --emergency --count=3
    \end{verbatim}
    
    \item \textbf{Queue Freeze:} Escalation queue will grow. Set queue priority to CRITICAL-only:
    \begin{verbatim}
    efm-cli escalation --mode=critical-only
    \end{verbatim}
    
    \item \textbf{Capsule Guidance:} Broadcast to capsules: ``Arbiters recovering, reduce escalations''
\end{enumerate}

\textbf{Recovery Criteria:}
\begin{itemize}
    \item Arbiter availability $> 50\%$
    \item Escalation queue $< 100$ pending
    \item No new Arbiter failures for 10K ticks
\end{itemize}

\textbf{Post-Incident:} Root cause analysis required. Check for common cause (resource exhaustion, coordinated attack, network partition).
\end{playbookbox}

\subsection{Catastrophic Failure: d-CTM Compromise}

\begin{playbookbox}[Runbook 2: d-CTM Integrity Compromised]
\textbf{Trigger:} d-CTM hash verification failure OR write latency $> 1000$ms sustained.

\textbf{Severity:} S5 (EMERGENCY)

\textbf{Immediate Actions:}
\begin{enumerate}
    \item \textbf{Halt All Operations:} This is a Layer 0 violation. Issue forest-wide FREEZE:
    \begin{verbatim}
    gardener forest --action=FREEZE --reason=DCTM_COMPROMISE
    \end{verbatim}
    
    \item \textbf{Isolate d-CTM:} Prevent further writes until integrity verified:
    \begin{verbatim}
    efm-cli dctm --mode=read-only
    \end{verbatim}
    
    \item \textbf{Integrity Check:} Run full verification against backup:
    \begin{verbatim}
    efm-cli dctm --verify --against=backup
    \end{verbatim}
    
    \item \textbf{If Corrupted:} Initiate rollback to last known-good state (see Appendix J \S8).
\end{enumerate}

\textbf{Recovery Criteria:}
\begin{itemize}
    \item d-CTM integrity verified (hash match)
    \item Write latency $< 100$ms
    \item Gardener sign-off on recovery
\end{itemize}

\textbf{Post-Incident:} Mandatory forensic replay (Appendix M). Identify attack vector or corruption source.
\end{playbookbox}

\subsection{Catastrophic Failure: Runaway Spawn}

\begin{playbookbox}[Runbook 3: Runaway Spawn Event]
\textbf{Trigger:} Spawn rate $> 2 \times R_{max}$ OR depth exceeds $D_{max}$ anywhere.

\textbf{Severity:} S4 (CRITICAL)

\textbf{Immediate Actions:}
\begin{enumerate}
    \item \textbf{Emergency Lockdown:} ASG should auto-trigger, but verify:
    \begin{verbatim}
    efm-cli spawn --check-lockdown
    \end{verbatim}
    
    \item \textbf{If Not Locked:} Manually trigger:
    \begin{verbatim}
    gardener spawn --lockdown --R_max=0 --tau=1.0
    \end{verbatim}
    
    \item \textbf{Identify Source:} Query d-CTM for spawn lineage:
    \begin{verbatim}
    efm-cli lineage --depth-exceeded --last=1000
    \end{verbatim}
    
    \item \textbf{Quarantine Branch:} Isolate the runaway lineage:
    \begin{verbatim}
    gardener quarantine --lineage=<ROOT_ID>
    \end{verbatim}
\end{enumerate}

\textbf{Recovery Criteria:}
\begin{itemize}
    \item Spawn rate $\leq R_{max}$
    \item All capsules $D \leq D_{max}$
    \item ASG parameters restored to operational bounds
\end{itemize}

\textbf{Post-Incident:} Review ASG configuration. Was $R_{max}$ appropriate? Adjust deployment profile if needed.
\end{playbookbox}

\subsection{Cascading Failure Decision Tree (Gap 7)}

\begin{warningbox}[Critical Operational Guidance]
Real-world failures cascade. When multiple symptoms appear simultaneously, use this decision tree to identify the \textbf{primary failure} and execute the correct runbook.
\end{warningbox}

\begin{playbookbox}[Runbook 0: Cascading Failure Diagnosis]
\textbf{Trigger:} Multiple failure symptoms detected simultaneously (e.g., Arbiter failures + escalation queue growth + d-CTM slowdown).

\textbf{The Problem:}
\begin{verbatim}
Tick 0:    Arbiter failure -> Escalation queue grows
Tick 1K:   Queue growth -> Memory pressure -> d-CTM writes slow
Tick 2K:   d-CTM slow -> Health monitoring delayed -> False readings
Tick 3K:   False health -> ASG reduces spawn -> System underutilized
Tick 4K:   Multiple symptoms: Which runbook?
\end{verbatim}

\textbf{Step 1: Identify Primary Failure}

Run diagnostics in this \textbf{exact priority order}:

\begin{enumerate}
    \item \textbf{d-CTM Integrity Check} (highest priority):
    \begin{verbatim}
    efm-cli dctm --verify-integrity
    \end{verbatim}
    \textit{If CORRUPTED} $\rightarrow$ \textbf{Runbook 2 IMMEDIATELY} (this is Layer 0)

    \item \textbf{Reflex Engine Health}:
    \begin{verbatim}
    efm-cli reflex --status
    \end{verbatim}
    \textit{If DOWN} $\rightarrow$ \textbf{EMERGENCY HALT} (Layer 0 violation)

    \item \textbf{Arbiter Availability}:
    \begin{verbatim}
    efm-cli arbiters --status
    \end{verbatim}
    \textit{If $< 50\%$} $\rightarrow$ \textbf{Runbook 1} (Arbiter recovery)

    \item \textbf{Spawn Rate Anomaly}:
    \begin{verbatim}
    efm-cli spawn --check
    \end{verbatim}
    \textit{If $> R_{max}$} $\rightarrow$ \textbf{Runbook 3} (spawn lockdown)
\end{enumerate}

\textbf{Step 2: Execute Primary Runbook}

Follow the procedures for whichever failure was identified first. Do NOT try to address secondary effects until primary is resolved.

\textbf{Step 3: Monitor Secondary Effects}

Every 1000 ticks, re-run Step 1 diagnostics:
\begin{itemize}
    \item If all checks pass $\rightarrow$ Recovery in progress
    \item If new failure detected $\rightarrow$ Escalate to Gardener (human decision needed)
    \item If same failure persists $\rightarrow$ Continue current runbook
\end{itemize}

\textbf{Step 4: Root Cause Analysis}

After recovery, mandatory forensic replay (Appendix M):
\begin{itemize}
    \item Which component failed first?
    \item How did failure propagate?
    \item How to prevent cascade in future?
\end{itemize}

\textbf{Decision Matrix:}
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Symptom Combination} & \textbf{Likely Primary} & \textbf{Action} \\
\midrule
Arbiter $\downarrow$ + Queue $\uparrow$ & Arbiter failure & Runbook 1 \\
d-CTM slow + Everything else & d-CTM issue & Runbook 2 \\
Spawn $\uparrow$ + Depth $\uparrow$ & Runaway spawn & Runbook 3 \\
All symptoms at once & d-CTM first & Runbook 2 then reassess \\
Reflex down + Anything & Reflex failure & EMERGENCY HALT \\
\bottomrule
\end{tabular}
\end{center}
\end{playbookbox}

\subsection{Partial System Failure}

\begin{playbookbox}[Playbook: Partial Arbiter Failure (20--50\%)]
\textbf{Trigger:} Arbiter availability between 20\% and 50\%.

\textbf{Severity:} S3 (HIGH)

\textbf{Note:} This is NOT ``all Arbiters failed'' (Runbook 1) but partial degradation.

\textbf{Steps:}
\begin{enumerate}
    \item \textbf{Assess Severity:}
    \begin{itemize}
        \item $>50\%$ available $\rightarrow$ Minor degradation, monitor only
        \item $20-50\%$ available $\rightarrow$ This playbook
        \item $<20\%$ available $\rightarrow$ Runbook 1 (emergency)
    \end{itemize}
    
    \item \textbf{Spawn Replacement Arbiters:}
    \begin{verbatim}
    gardener spawn-arbiters --count=<FAILED_COUNT/2>
    \end{verbatim}
    
    \item \textbf{Redistribute Load:}
    \begin{verbatim}
    efm-cli arbiters --rebalance
    \end{verbatim}
    
    \item \textbf{Investigate Common Cause:}
    \begin{itemize}
        \item Network partition? Check connectivity between Arbiter nodes
        \item Resource exhaustion? Check memory/CPU on failed nodes
        \item Coordinated attack? Check for suspicious patterns in d-CTM
    \end{itemize}
    
    \item \textbf{Alert Gardener:} Partial failures often indicate emerging problems
\end{enumerate}

\textbf{Recovery Criteria:}
\begin{itemize}
    \item Arbiter availability $> 90\%$
    \item No new failures for 5K ticks
    \item Escalation queue stable
\end{itemize}
\end{playbookbox}

\subsection{Safe Change Workflow}

All parameter changes follow this workflow:

\begin{enumerate}
    \item \textbf{Propose:} Submit change request with justification
    \item \textbf{Validate:} For $\tau$/$\lambda$ changes, run Monotonic Sensitivity check (Vol.~I Theorem 3.1)
    \item \textbf{Approve:} Gardener signs change (or Arbiter for Threshold Governance)
    \item \textbf{Log:} Change recorded to d-CTM with ZK-SP proof
    \item \textbf{Monitor:} Watch affected metrics for $T_{observation}$ ticks
    \item \textbf{Rollback:} If regression detected, revert via same workflow
\end{enumerate}

\begin{warningbox}[Never Modify Directly]
The following are \textbf{immutable} and cannot be changed at runtime:
\begin{itemize}
    \item Vault Commandments (Layer 0)
    \item Reflex-Core rules (Layer 0.5)
    \item Genesis root hash
    \item d-CTM append-only property
\end{itemize}
Changes to these require full system rebuild.
\end{warningbox}

\section{Governance and Audits}

\subsection{Gardener Responsibilities}

\begin{enumerate}
    \item \textbf{Override Authority:} Can override any d-CAM verdict (subject to Vault constraints)
    \item \textbf{DCG Review:} Review Deliberation Context Graphs for high-impact decisions (Vol.~II \S2.3)
    \item \textbf{Micro-Heuristic Approval:} Sign off on Reflex-Heuristic updates derived from Arbiter precedent
    \item \textbf{Conflict Arbitration:} Resolve disputes between safety and performance
    \item \textbf{Audit Cooperation:} Provide d-CTM access for compliance reviews
\end{enumerate}

\subsection{Incident Reconstruction}

To reconstruct any incident from audit trail:

\begin{enumerate}
    \item \textbf{Identify Verdict:} Query d-CTM by capsule ID, timestamp, or verdict type
    \item \textbf{Retrieve DCG:} Use \texttt{dcg\_hash} to fetch full reasoning graph (Vol.~II Definition 2.3)
    \item \textbf{Verify ZK-SP:} Confirm proof chain is valid (tamper detection)
    \item \textbf{Trace Lineage:} Follow Vault Hash Chain to genesis for affected capsules
    \item \textbf{Document:} Produce audit report showing invariant compliance
\end{enumerate}
